    var TipHover = function (tgerDiv, showDiv, tgerDivClass, showDivClass, timerSeconds) {
        var curTgerDivClass = C.$(tgerDiv).className;
        var curshowDivClas = C.$(showDiv).className;
        this.init = function () {
            C.bind(tgerDiv, 'mousemove', tipMove);
            C.bind(showDiv, 'mousemove', tip_msgMove);
            if (typeof C.$(tgerDiv).onmouseleave != 'undefined') {
                C.bind(tgerDiv, 'mouseleave', tipOut);
                C.bind(showDiv, 'mouseleave', tip_msgOut);
            }
            else {
                C.bind(tgerDiv, 'mouseout', tipOut);
                C.bind(showDiv, 'mouseout', tip_msgOut);

        var isuserMove = true;
        var showtimer = null;
        var tipMove = function () {
            if (showtimer != null) return;
            showtimer = setTimeout(function () {
                if (C.$(showDiv).style.display == 'none') {
                    C.$(showDiv).style.display = 'block';
                    if (typeof tgerDivClass != 'undefined' && tgerDivClass.length > 0) C.$(tgerDiv).className = tgerDivClass;
                    if (typeof showDivClass != 'undefined' && showDivClass.length > 0) C.$(showDiv).className = showDivClass;
                    showtimer = null;
                }
            }, 150);
            if (timer != null) { clearTimeout(timer); timer = null; }
        };
        var timer = null;
        var tipOut = function () {
            if (C.$(showDiv).style.display == 'none') { clearTimeout(showtimer); showtimer = null; return; }
            if (timer != null) { clearTimeout(timer); }
            timer = setTimeout(function () {
                if (isuserMove) {
                    C.$(showDiv).style.display = 'none';
                    C.$(tgerDiv).className = curTgerDivClass;
                    C.$(showDiv).className = curshowDivClas;
                    clearTimeout(timer);
                    timer = null;
                    clearTimeout(showtimer);
                    showtimer = null;
                }
            }, 100);
        };
        var tip_msgMove = function () {
            isuserMove = false;
        };
        var tip_msgOut = function () {
            isuserMove = true;
            tipOut();
        };
    };
